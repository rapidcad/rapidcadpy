@staticmethod
def from_vector(vec, start_point=None, is_numerical=False):
    mid_point = np.array([vec[2], vec[3]])
    end_point = np.array([vec[0], vec[1]])
    if is_numerical:
        mid_point = mid_point / QUANTIZATION_SCALE
        end_point = end_point / QUANTIZATION_SCALE
    return Arc(
        start_point=Vertex(*start_point),
        mid_point=Vertex(*mid_point),
        end_point=Vertex(*end_point),
    )

```python
def numericalize(self, n=256):
    self.start_point = (
        (self.start_point * n).round().clip(min=0, max=n - 1).astype(int)
    )
    self.mid_point = (self.mid_point * n).round().clip(min=0, max=n - 1).astype(int)
    self.end_point = (self.end_point * n).round().clip(min=0, max=n - 1).astype(int)
    self._is_numerical = True
```
