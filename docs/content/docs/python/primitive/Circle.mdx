def flip(self, axis):
    if axis == "x":
        self.center[1] = -self.center[1]
    elif axis == "y":
        self.center[0] = -self.center[0]
    elif axis == "xy":
        self.center = self.center * -1
    else:
        raise ValueError("axis = {}".format(axis))

def numericalize(self, n=256):
    self.center = (self.center * n).round().clip(min=0, max=n - 1).astype(int)
    self.radius = max(1, min(int(round(self.radius * n)), n - 1))

```python
def sample_points(self, n=32):
    angles = np.linspace(0, 2 * np.pi, n, endpoint=False)
    x = self.center.x + self.radius * np.cos(angles)
    y = self.center.y + self.radius * np.sin(angles)
    return np.column_stack([x, y])
```
