def __init__(
    self,
    start_point: Union[Iterable, Vertex],
    end_point: Union[Iterable, Vertex],
    id: Optional[uuid.UUID] = None,
    name: str = "line",
):
    super(Line, self).__init__(id, name)
    if isinstance(start_point, Vertex):
        self.start_point = start_point
    else:
        if len(start_point) > 2:
            raise
        self.start_point = Vertex(x=float(start_point[0]), y=float(start_point[1]))

    if isinstance(end_point, Vertex):
        self.end_point = end_point
    else:
        if len(end_point) > 2:
            raise
        self.end_point = Vertex(x=float(end_point[0]), y=float(end_point[1]))
```

@staticmethod
def from_fs_dict(entity_dict):
    entity_dict = entity_dict["message"]
    geometry_dict = entity_dict["geometry"]["message"]
    pnt_x = geometry_dict["pntX"]
    pnt_y = geometry_dict["pntY"]
    dir_x = geometry_dict["dirX"]
    dir_y = geometry_dict["dirY"]
    start_param = entity_dict["startParam"]
    end_param = entity_dict["endParam"]

    startX = pnt_x + start_param * dir_x
    startY = pnt_y + start_param * dir_y

    endX = pnt_x + end_param * dir_x
    endY = pnt_y + end_param * dir_y
    return Line([startX, startY], [endX, endY])
```

```python
@staticmethod
def from_json(stat, vertices=None):
    if vertices is None:
        start_point = stat["start"]
        end_point = stat["end"]
        return Line(
            np.array([start_point["x"], start_point["y"]]),
            np.array([end_point["x"], end_point["y"]]),
        )
    else:
        assert stat["type"] == "Line"
        start_point = vertices[stat["start"]]
        end_point = vertices[stat["end"]]
        return Line(start_point, end_point)
```

def to_occ(self):
    from OCC.Core.BRepBuilderAPI import BRepBuilderAPI_MakeEdge
    from OCC.Core.gp import gp_Pnt

    p1 = gp_Pnt(
        self.start_point.x, self.start_point.y, getattr(self.start_point, "z", 0.0)
    )
    p2 = gp_Pnt(
        self.end_point.x, self.end_point.y, getattr(self.end_point, "z", 0.0)
    )
    return BRepBuilderAPI_MakeEdge(p1, p2).Edge()
