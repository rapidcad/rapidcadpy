title: Workplane

A fluent API workplane that provides CadQuery-like syntax for CAD operations.

This class maintains a working plane and allows chaining operations to build
complex 3D models using a simple, intuitive syntax.

def line(self, x: float, y: float) -> "Workplane":
    """
    Create a line from current position to the specified coordinates.

    Args:
        x: End X coordinate
        y: End Y coordinate

    Returns:
        Self for method chaining
    """
    from .cad_types import Vertex
    from .primitive import Line

    end_point = Vertex(x, y)
    line = Line(start_point=self._current_position, end_point=end_point)
    self._pending_shapes.append(line)
    self._current_position = end_point
    return self

```python
def rect(self, width: float, height: float, centered: bool = True) -> "Workplane":
    """
    Create a rectangle.

    Args:
        width: Rectangle width
        height: Rectangle height
        centered: If True, center at current position; if False, start at current position

    Returns:
        Self for method chaining
    """
    from .cad_types import Vertex
    from .primitive import Line

    if centered:
        x_offset = width / 2
        y_offset = height / 2
        start_x = self._current_position.x - x_offset
        start_y = self._current_position.y - y_offset
    else:
        start_x = self._current_position.x
        start_y = self._current_position.y

    # Create rectangle as four lines
    p1 = Vertex(start_x, start_y)
    p2 = Vertex(start_x + width, start_y)
    p3 = Vertex(start_x + width, start_y + height)
    p4 = Vertex(start_x, start_y + height)

    self._pending_shapes.extend(
        [
            Line(start_point=p1, end_point=p2),
            Line(start_point=p2, end_point=p3),
            Line(start_point=p3, end_point=p4),
            Line(start_point=p4, end_point=p1),
        ]
    )

    return self
```

def add(self, other: "Workplane") -> "SketchBuilder":
    """
    Combine this workplane with another and return a SketchBuilder.

    Args:
        other: Another workplane to combine with

    Returns:
        SketchBuilder for further operations
    """
    # Combine pending shapes from both workplanes
    all_shapes = self._pending_shapes + other._pending_shapes
    return SketchBuilder(self.plane, all_shapes)

def extrude(self, distance: float, operation: str = "NewBodyFeatureOperation"):
    """
    Extrude the pending shapes and return a CAD object.

    Args:
        distance: Extrusion distance
        operation: Extrusion operation type

    Returns:
        CAD object with the extruded feature
    """
    # Create a sketch builder and extrude
    builder = SketchBuilder(self.plane, self._pending_shapes)
    return builder.extrude(distance, operation)
