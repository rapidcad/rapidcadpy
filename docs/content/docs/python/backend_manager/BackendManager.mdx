```python
def get_available_backends(self) -> Dict[str, bool]:
    """
    Get all registered backends and their availability.

    Returns:
        Dictionary mapping backend names to availability status
    """
    result = {}
    for name, backend_class in self._backends.items():
        try:
            backend = backend_class()
            result[name] = backend.is_available()
        except:
            result[name] = False
    return result
```

```python
def set_backend(self, name: str) -> bool:
    """
    Set the current backend.

    Args:
        name: Name of the backend to use

    Returns:
        True if backend was successfully set, False otherwise
    """
    name = name.lower()
    if name not in self._backends:
        print(f"Backend '{name}' is not registered")
        return False

    try:
        backend_class = self._backends[name]
        backend = backend_class()

        if not backend.is_available():
            print(f"Backend '{name}' is not available")
            return False

        self._current_backend = backend
        print(f"Set backend to: {backend.get_name()}")
        return True

    except Exception as e:
        print(f"Failed to initialize backend '{name}': {e}")
        return False
```

```python
def _auto_select_default(self):
    """Automatically select the best available backend."""
    # Priority order: Inventor (if available), then OCC
    priority_order = ["inventor", "occ"]

    for backend_name in priority_order:
        if self.set_backend(backend_name):
            self._default_backend = backend_name
            break

    if not self._current_backend:
        print("Warning: No CAD backend is available")
```

```python
def export_cad(self, cad, filename: Optional[str] = None, **kwargs) -> Any:
    """
    Export CAD using the current backend.

    Args:
        cad: CAD object to export
        filename: Optional filename
        **kwargs: Backend-specific options

    Returns:
        Backend-specific result
    """
    if not self._current_backend:
        raise RuntimeError("No backend is currently set")

    return self._current_backend.export_cad(cad, filename, **kwargs)
```
