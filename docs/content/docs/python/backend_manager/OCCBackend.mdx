def _initialize_integration(self):
    """Initialize the OCC integration if available."""
    try:
        from .integrations.occ_integration import OccIntegration

        self._integration = OccIntegration()
    except Exception as e:
        print(f"Warning: Could not initialize OCC backend: {e}")
        self._integration = None

```python
def export_cad(self, cad, filename: Optional[str] = None, **kwargs) -> Any:
    """Export CAD to OCC format (STEP, BREP, etc.)."""
    if not self.is_available():
        raise RuntimeError("OCC backend is not available")

    if filename:
        if filename.endswith(".step") or filename.endswith(".stp"):
            return cad.to_step(filename)
        elif filename.endswith(".brep"):
            return cad.to_brep(filename)
        elif filename.endswith(".stl"):
            return cad.to_stl(filename)
        else:
            # Default to STEP
            return cad.to_step(filename)
    else:
        # Return the OCC shape object
        return cad.to_occ()
```

```python
def is_available(self) -> bool:
    """Check if OCC backend is available."""
    if self._integration is None:
        return False
    try:
        # Check if the CAD object has OCC methods
        from .cad import Cad

        return hasattr(Cad, "to_occ") and hasattr(Cad, "to_step")
    except:
        return False
```
