Backend system for rapidcadpy - allows switching between different CAD backends.

This module provides a unified interface for different CAD backends (Inventor, OCC, etc.)
and allows users to easily switch between them for the fluent API.

```python
def _add_backend_methods_to_cad():
    """Add backend methods to the CAD class."""
    from .cad import Cad

    def to_inventor(
        self, filename: Optional[str] = None, close_on_save: bool = False, **kwargs
    ):
        """
        Export this CAD object to Inventor using the current backend.

        Args:
            filename: Optional filename to save to
            close_on_save: Whether to close the document after saving
            **kwargs: Additional backend-specific options
        """
        # Temporarily switch to Inventor backend if not current
        original_backend = get_current_backend()

        try:
            if original_backend != "inventor":
                if not _backend_manager.set_backend("inventor"):
                    raise RuntimeError("Inventor backend is not available")

            return _backend_manager.export_cad(
                self, filename, close_on_save=close_on_save, **kwargs
            )
        finally:
            # Restore original backend if we changed it
            if original_backend and original_backend != "inventor":
                _backend_manager.set_backend(original_backend)

    def to_backend(self, backend_name: str, filename: Optional[str] = None, **kwargs):
        """
        Export this CAD object using a specific backend.

        Args:
            backend_name: Name of the backend to use
            filename: Optional filename to save to
            **kwargs: Backend-specific options
        """
        # Temporarily switch to the requested backend
        original_backend = get_current_backend()

        try:
            if not _backend_manager.set_backend(backend_name):
                raise RuntimeError(f"Backend '{backend_name}' is not available")

            return _backend_manager.export_cad(self, filename, **kwargs)
        finally:
            # Restore original backend
            if original_backend:
                _backend_manager.set_backend(original_backend)

    def export(self, filename: Optional[str] = None, **kwargs):
        """
        Export this CAD object using the current backend.

        Args:
            filename: Optional filename to save to
            **kwargs: Backend-specific options
        """
        return _backend_manager.export_cad(self, filename, **kwargs)

    # Add methods to CAD class
    Cad.to_inventor = to_inventor
    Cad.to_backend = to_backend
    Cad.export = export
```
