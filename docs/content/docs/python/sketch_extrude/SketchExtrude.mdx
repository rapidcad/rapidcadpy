```python
def to_python(self, index: Optional[int] = 0) -> str:
    variables = []
    declarations = []
    sketch_variables = []
    for sketch_index, sketch in enumerate(self.sketch):
        sketch_variables.append(f"sketch_{sketch_index}")
        declarations.append(f"{sketch.to_python(sketch_index)}")
    if self.extrude:
        variables.append(f"extrude_{index}")
        declarations.append(f"extrude_{index} = {self.extrude.to_python(index)}")

    code = "\n".join(declarations)
    sketch_variable_list = ", ".join(sketch_variables)
    sketch_variable_list = "[" + sketch_variable_list + "]"
    code += "\nsketch_extrude_{} = SketchExtrude(sketch={}, extrude={})\n".format(
        index, sketch_variable_list, variables[0] if variables else "None"
    )
    return code
```

```python
def plot(self, title="Sketch Extrude", bounding_box=False):
    fig = go.Figure()
    for sketch in self.sketch:
        for trace in sketch.plot(return_fig=True):
            if trace is not None:
                fig.add_trace(trace)
    # add the bounding box
    if bounding_box:
        bbox_min, bbox_max = self.bbox
        x = [bbox_min[0], bbox_max[0], bbox_max[0], bbox_min[0], bbox_min[0]]
        y = [bbox_min[1], bbox_min[1], bbox_max[1], bbox_max[1], bbox_min[1]]
        fig.add_trace(go.Scatter(x=x, y=y, mode="lines", name="Bounding Box"))
    fig.update_layout(
        title=title,
        scene=dict(aspectmode="cube"),
        xaxis=dict(scaleanchor="y", title="X Axis"),
        yaxis=dict(title="Y Axis"),
        showlegend=False,
        autosize=True,
    )
    fig.show()
```
